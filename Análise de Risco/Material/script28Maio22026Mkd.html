I-Mostrando os efeitos do coeficiente de correlação

mu=c(0,0)
cCor=c(0.1,0.5,0.95,-0.1,-0.5,-0.95)

par(mfrow=c(2,3))
for (icCor in 1:6){
  sigma=matrix(c(1,cCor[icCor],cCor[icCor],1),nrow=2, byrow=T)
  df<-mvrnorm(1000,mu=mu,Sigma=sigma,tol=1e-3)
  mainLbl=paste('rho= ', cCor[icCor])
  plot(df[,1],df[,2], main=mainLbl )
  abline(h=0,v=0)
}

II-Gerando triangulares correlacionadas

Passo 1: Gerar normais reduzidas correlacionadas

par(mfrow=c(1,3))
nSim=1000
#correlation parameters
mu=c(0,0)
rho=0.5

#triangular parameters
min=5
max=10
mp=7

sigma=matrix(c(1,rho,rho,1),nrow=2, byrow=T)
sigma
##      [,1] [,2]
## [1,]  1.0  0.5
## [2,]  0.5  1.0
df<-mvrnorm(nSim, mu,sigma)
colnames(df)<-c('x1','x2')
head(df)

Passo 2: Obter uniformes(0,1) correlacionadas

par(mfrow=c(1,3))
u1<-pnorm(df[,1])

hist(u1,main='Histogram of cumulative of x1')
u2<-pnorm(df[,2])

hist(u2,main='Histogram of cumulative of x1')
plot(u1,u2, main=('Duas uniformes correlacionadas'), xlab='u1',yla='u2')
